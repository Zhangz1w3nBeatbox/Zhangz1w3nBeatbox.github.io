class Solution {
    public int maxProfit(int k, int[] prices) {
        int[] buy = new int[k+1];
        int[] sale = new int[k+1];

        Arrays.fill(buy,Integer.MIN_VALUE);
        Arrays.fill(sale,0);

        for(int p:prices){

            for(int i=1;i<=k;++i){

                buy[i] = Math.max(buy[i],sale[i-1]-p);

                sale[i] = Math.max(sale[i],buy[i]+p);
                
            }

        }

        return sale[k];
